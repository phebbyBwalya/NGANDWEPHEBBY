    <h2>Score to Grade Converter</h2>
    <label>Score (0–100): <input type="number" id="score" /></label>
    <div class="error" id="grade-error"></div>
    <button onclick="computeGrade()">Compute</button>
    <button onclick="resetGrade()">Reset</button>
    <div class="result" id="grade-result"></div>
  </div>

  <script>
    function solveQuadratic() {
      const a = parseFloat(document.getElementById("a").value);
      const b = parseFloat(document.getElementById("b").value);
      const c = parseFloat(document.getElementById("c").value);
      const error = document.getElementById("quad-error");
      const result = document.getElementById("quad-result");
      error.textContent = "";
      result.textContent = "";

      if (isNaN(a) || isNaN(b) || isNaN(c)) {
        error.textContent = "Please enter valid numbers for a, b, and c.";
        return;
      }
      if (a === 0) {
        error.textContent = "Coefficient 'a' must not be zero.";
        return;
      }

      const D = b * b - 4 * a * c;
      let nature = "";
      let roots = "";

      if (D > 0) {
        nature = "Two distinct real roots";
        const r1 = (-b + Math.sqrt(D)) / (2 * a);
        const r2 = (-b - Math.sqrt(D)) / (2 * a);
        roots = `x₁ = ${r1.toFixed(3)}, x₂ = ${r2.toFixed(3)}`;
      } else if (D === 0) {
        nature = "One real repeated root";
        const r = -b / (2 * a);
        roots = `x = ${r.toFixed(3)}`;
      } else {
        nature = "Two complex conjugate roots";
        const real = (-b / (2 * a)).toFixed(3);
        const imag = (Math.sqrt(-D) / (2 * a)).toFixed(3);
        roots = `x₁ = ${real} + ${imag}i, x₂ = ${real} - ${imag}i`;
      }

      result.textContent = `Discriminant: ${D.toFixed(3)}\n${nature}\n${roots}`;
    }

    function resetQuadratic() {
      document.getElementById("a").value = "";
      document.getElementById("b").value = "";
      document.getElementById("c").value = "";
      document.getElementById("quad-error").textContent = "";
      document.getElementById("quad-result").textContent = "";
    }

    function computeGrade() {
      const score = parseInt(document.getElementById("score").value);
      const error = document.getElementById("grade-error");
      const result = document.getElementById("grade-result");
      error.textContent = "";
      result.textContent = "";

      if (isNaN(score) || score < 0 || score > 100) {
        error.textContent = "Please enter a valid score between 0 and 100.";
        return;
      }

      let grade = "";
      if (score >= 85) grade = "A+";
      else if (score >= 75) grade = "A";
      else if (score >= 65) grade = "B+";
      else if (score >= 60) grade = "B";
      else if (score >= 55) grade = "C+";
      else if (score >= 50) grade = "C";
      else grade = "D";

      result.textContent = `Score ${score} → Grade ${grade}`;
    }

    function resetGrade() {
      document.getElementById("score").value = "";
      document.getElementById("grade-error").textContent = "";
      document.getElementById("grade-result").textContent = "";
    }
  </script>
